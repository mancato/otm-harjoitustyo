2   package logic;
3   
4   import javafx.scene.shape.Polygon;
5   import javafx.scene.paint.Color;
6   public class Ship extends PolygonObject {
7   
8       public int lives;
9       public int spawnCounter;
10      public int turretTemp;
11  
12      public Ship(double x0, double y0, double scale) {
13          this.lives = 3; //NUMBER OF INITIAL LIVES
14          this.setTranslateX(x0);
15          this.setTranslateY(y0);
16          this.setFill(Color.TRANSPARENT);
17          this.setStroke(Color.WHITE);
18          this.getPoints().addAll(new Double[] { //SHIP FORM
19              scale * 6.0, 0.0,
20              -scale * 6.0, -scale * 4.0,
21              -scale * 3.0, 0.0,
22              -scale * 6.0, scale * 4.0
23          });
24          this.spawnCounter = -1; //FOR DELAYING SPAWN TIME, -1 MEANS SHIP IS IN PLAY
25          this.turretTemp = 0;
26      }
27      public void accelerate(double thrust) { //ACCELERATING THE SHIP
28          vX += thrust * Math.cos(Math.toRadians(this.getRotate())) - vX * 0.007;
29          vY += thrust * Math.sin(Math.toRadians(this.getRotate())) - vY * 0.007;
30      }
31      public void rotate(double angle) { //ROTATING THE SHIP
32          double a = this.getRotate() + angle;
33          if (a > 360.0) {
34              this.setRotate(a - 360.0);
35          } else if (a < 0.0) {
36              this.setRotate(a + 360.0);
37          } else {
38              this.setRotate(a);
39          }
40      }
41      public void spawn() { //SPAWNING A NEW SHIP
42          this.vX = 0.0;
43          this.vY = 0.0;
44          this.setTranslateX(320.0);
45          this.setTranslateY(240.0);
46          this.spawnCounter = -1;
47      }
48  }

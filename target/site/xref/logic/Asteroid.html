4   import javafx.scene.shape.Polygon;
5   import javafx.scene.paint.Color;
6   import java.util.Random;
7   
8   public class Asteroid extends PolygonObject {
9       
10      public double size;
11      Random rnd;
12      
13      public Asteroid(int n, double size, double x0, double y0) {
14          this.size = size;
15          this.rnd = new Random();
16          this.setTranslateX(x0);
17          this.setTranslateY(y0);
18          this.setFill(Color.TRANSPARENT);
19          this.setStroke(Color.WHITE);
20          this.vX = Math.pow(size, -0.7) * (2.0 * rnd.nextDouble() - 1.0); //RANDOM SPEED
21          this.vY = Math.pow(size, -0.7) * (2.0 * rnd.nextDouble() - 1.0);
22          this.getPoints().addAll(initialize(n)); //SET ASTEROID FORM       
23      }
24      Double[] initialize(int n) { //GENERATE RANDOM SURFACE FOR ASTEROID, n=AMOUNT OF VERTICES
25          double rMin = size * 30.0;
26          double rMax = size * (rMin + 25.0);
27          double radius = 0.0;
28          double angle = 0.0;
29          Double[] p = new Double[2 * n];
30          for (int i = 0; i < 2 * n - 1; i += 2) {
31              radius = rMax * rnd.nextDouble() + rMin;
32              angle = (rnd.nextDouble() + i) * Math.PI / n;;
33              p[i] = radius * Math.cos(angle);
34              p[i + 1] = radius * Math.sin(angle);
35          }
36          return p;
37      }

39  }

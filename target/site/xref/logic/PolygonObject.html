4   import javafx.scene.shape.Polygon;
5   
6   public class PolygonObject extends Polygon { //THE BASE CLASS FOR A MOVABLE POLYGONAL OBJECT
7       
8       public double vX;
9       public double vY;
10  
11      public void translate(double dx, double dy) { //MOVE BY AMOUNT DX,DY
12          this.setTranslateX(this.getTranslateX() + dx);
13          this.setTranslateY(this.getTranslateY() + dy);        
14      }
15      public void checkBounds() { //SCREEN WRAPPING FEATURE
16          if (this.getTranslateX() > 640.0) {
17              translate(-640.0, 0.0);
18              return;
19          }
20          if (this.getTranslateY() > 480.0) {
21              translate(0.0, -480.0);
22              return; 
23          }
24          if (this.getTranslateX() < 0.0) {
25              translate(640.0, 0.0);
26              return;          
27          }
28          if (this.getTranslateY() < 0.0) {
29              translate(0.0, 480.0);
30              return; 
31          }
32      }
33  }

6   package ui;
7   import logic.Ammo;
8   import logic.Asteroid;
9   import logic.AsteroidField;
10  import logic.Ship;
11  
12  import java.util.ArrayList;
13  import javafx.application.Application;
14  import javafx.stage.Stage;
15  import javafx.animation.AnimationTimer;
16  import javafx.scene.shape.Shape;
17  
18  public class Main extends Application {
19      int score;
20      boolean hold;
21      @Override
22      public void start(Stage stage) throws Exception {
23          //int score = 0;
24  
25          Interface ui = new Interface(stage, "ASTEROIDS", 640, 480);
26          AsteroidField asteroids = new AsteroidField();
27          Ship ship = new Ship(320.0, 240.0, 3.0);
28          ArrayList<Ammo> ammos = new ArrayList<Ammo>();
29  
30          ui.setScore(0);
31          ui.setLives(3);
32          ui.addShape(ship);
33          //THE GAME LOOP
34          AnimationTimer timer = new AnimationTimer() {
35              @Override
36              public void handle(long now) {
37                  //SPAWN MORE ASTEROIDS
38                  if (asteroids.isEmpty()) {
39                      if (asteroids.asteroidTimer > 0) {
40                          asteroids.asteroidTimer--;
41                      } else {        
42                          asteroids.spawn(asteroids.level, ship.getTranslateX(), ship.getTranslateY());
43                          for (Asteroid a: asteroids) {
44                              ui.addShape(a);
45                          }
46                          if (asteroids.level < 10) {
47                              asteroids.level++;
48                          }
49                      }
50                  }
51                  //DELAY FOR SPAWNING A NEW SHIP, spawnCounter = -1 MEANS SHIP IS IN PLAY
52                  if (ship.spawnCounter > 0) {
53                      ship.spawnCounter--;
54                  } else if (ship.spawnCounter == 0) {
55                      ship.spawn();
56                  }
57                  // SHIP CONTROLS & MOVEMENT
58                  if (ship.lives > 0 && ship.spawnCounter == -1) {
59                      double thrust = 0;              
60                      if (ui.up) {
61                          thrust += 0.08;
62                      } 
63                      if (ui.right) {
64                          ship.rotate(5.0);
65                      }
66                      if (ui.left) {
67                          ship.rotate(-5.0);
68                      }
69                      ship.accelerate(thrust);
70                      ship.checkBounds();
71                      ship.translate(ship.vX, ship.vY);
72                  }
73                  //SHOOT!
74                  if (ship.turretTemp > 0) {
75                      ship.turretTemp--;
76                  }
77                  if (!hold && ui.spacebar && ship.turretTemp + ship.spawnCounter == -1 && ship.lives > 0) {
78                      ammos.add(new Ammo(ship.getTranslateX(), ship.getTranslateY(),
79                          10.0 * Math.cos(Math.toRadians(ship.getRotate())) + ship.vX,
80                          10.0 * Math.sin(Math.toRadians(ship.getRotate())) + ship.vY)
81                      );
82                      ui.addShape(ammos.get(ammos.size() - 1));
83                      ship.turretTemp = 10;                    
84                  }
85                  if (ui.spacebar) {
86                      hold = true; //FOR PREVENTION OF SHOOTING sarrrrrrrrja
87                  } else {
88                      hold = false;
89                  }
90                  //MOVEMENT OF AMMUNITION
91                  for (int i = 0; i < ammos.size(); i++) {
92                      ammos.get(i).checkBounds();
93                      ammos.get(i).translate(ammos.get(i).vX, ammos.get(i).vY);
94                      ammos.get(i).range--;
95                      if (ammos.get(i).range == 0) {
96                          ui.removeShape(ammos.get(i));
97                          ammos.remove(i);
98                      }
99                  }
100                 //ASTEROID MOVEMENT & COLLISION CHECK WITH SHIP
101                 for (int i = 0; i < asteroids.size(); i++) {
102                     Asteroid a = asteroids.get(i);
103                     a.checkBounds();
104                     a.translate(a.vX, a.vY);
105                     //COLLISION CHECK
106                     if (Shape.intersect(a, ship).getBoundsInLocal().getWidth() != -1) {
107                         ship.lives--; //UPDATE LIFE COUNT
108                         ui.setLives(ship.lives);
109                         //GAME OVER, NO LIVES LEFT
110                         if (ship.lives == 0) {
111                             ship.setTranslateX(1000.0);
112                             ship.vX = 0.0;
113                             ship.vY = 0.0;
114                             ui.removeShape(ship);
115                             ui.gameOver();
116                         } else { //LIVES LEFT, SET DELAY FOR SPAWNING A NEW SHIP
117                             ship.spawnCounter = 50;
118                             ship.setTranslateX(1000.0);
119                         }  
120                     }
121                     for (int j = 0; j < ammos.size(); j++) {
122                         if (Shape.intersect(a, ammos.get(j)).getBoundsInLocal().getWidth() != -1) {
123                             ui.removeShape(ammos.get(j));
124                             ammos.remove(j);
125                             score += (int) (100 / a.size); //UPDATE SCORE WITH HIT
126                             ui.setScore(score);
127                             //SPAWN 2 NEW ASTEROIDS IF ASTEROID SIZE IS LARGE ENOUGH
128                             if (a.size > 0.25) {
129                                 asteroids.add(
130                                     new Asteroid(13, a.size / 2.0
131                                     , a.getTranslateX(), a.getTranslateY())
132                                 );
133                                 asteroids.add(
134                                     new Asteroid(13, a.size / 2.0
135                                     , a.getTranslateX(), a.getTranslateY())
136                                 );
137                                 ui.addShape(asteroids.get(asteroids.size() - 1));
138                                 ui.addShape(asteroids.get(asteroids.size() - 2));
139                             }
140                             //REMOVE PREVIOUS ASTEROID
141                             ui.removeShape(a);
142                             asteroids.remove(i); 
143                         }
144                     }
145                 }              
146             }
147         };
148         timer.start();
149     }
150 
151     public static void main(String args[]) {
152         Application.launch(args);        
153     }
154 }

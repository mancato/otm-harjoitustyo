2   package ui;
3   
4   import javafx.application.Platform;
5   import javafx.stage.Stage;
6   import javafx.scene.Scene;
7   import javafx.scene.layout.Pane;
8   import javafx.scene.input.KeyEvent;
9   import javafx.scene.paint.Color;
10  import javafx.scene.text.Text;
11  import javafx.scene.text.Font;
12  import javafx.scene.shape.Shape;
13  import javafx.scene.shape.Polygon;
14  import javafx.event.EventHandler;
15  
16  public class Interface{
17  
18      Stage stage;
19      Scene scene;
20      Pane pane;
21      Text score;
22      Text lives;
23      Polygon lifeSymbol;
24      boolean up,left,right,spacebar;
25  
26      public Interface(Stage stage, String title, int width, int height){
27          this.stage = stage;
28          this.pane = new Pane();
29          pane.setStyle("-fx-background-color: black;");
30          this.scene = new Scene(pane,640,480);
31          //DRAW SCORE
32          score = new Text(20,30,String.valueOf(0)); //GAME SCORE AS TEXT
33          score.setFill(Color.WHITE);
34          score.setFont(Font.font("Tlwg Typo", 30.0));
35          pane.getChildren().add(score);       
36          //DRAW NUMBER OF LIVES
37          lives = new Text(45.0,65.0,"x");
38          lives.setFill(Color.WHITE);
39          lives.setFont(Font.font("Tlwg Typo", 20.0));
40          pane.getChildren().add(lives);
41          //DRAW LIVES SYMBOL
42          Polygon lifeSymbol = new Polygon();
43          lifeSymbol.setFill(Color.TRANSPARENT);
44          lifeSymbol.setStroke(Color.WHITE);
45          lifeSymbol.setTranslateX(30.0);
46          lifeSymbol.setTranslateY(60.0);
47          lifeSymbol.getPoints().addAll(new Double[]{ //FORM OF SHIP SYMBOL
48              0.0, -12.0,
49              -8.0, 12.0,
50              0.0, 6.0,
51              8.0, 12.0
52          });
53          pane.getChildren().add(lifeSymbol);
54          //SETUP        
55          stage.setScene(scene);
56          stage.setTitle(title);
57          stage.setMinWidth(width);
58          stage.setMinHeight(height);
59          stage.setMaxWidth(width);
60          stage.setMaxHeight(height);
61          stage.setOnCloseRequest(e -> Platform.exit());
62          stage.show();
63          //KEY INPUT - KEY PRESS
64          scene.setOnKeyPressed(new EventHandler<KeyEvent>() {
65              @Override
66              public void handle(KeyEvent event) {
67                  switch (event.getCode()) {
68                      case UP:    up = true; break;
69                      case LEFT:  left = true; break;
70                      case RIGHT: right = true; break;
71                      case SPACE: spacebar = true; break;
72                  }
73              }
74          });
75          //KEY INPUT - RELEASE
76          scene.setOnKeyReleased(new EventHandler<KeyEvent>() { 
77              @Override
78              public void handle(KeyEvent event) {
79                  switch (event.getCode()) {
80                      case UP:    up = false; break;
81                      case LEFT:  left = false; break;
82                      case RIGHT: right = false; break;
83                      case SPACE: spacebar = false; break;
84                  }
85              }
86          });          
87      }
88  
89      void addShape(Shape shape){ //FOR DISPLAYING OBJECTS ON SCREEN
90          this.pane.getChildren().add(shape);
91      }
92  
93      void removeShape(Shape shape){ //FOR REMOVING OBJECTS FROM SCREEN
94          this.pane.getChildren().remove(shape);
95      }
96      
97      void setScore(int score){ //UPDATE SCORE SHOWN ON SCREEN
98              this.score.setText(String.valueOf(score));
99      }
100 
101     void setLives(int lives){ //UPDATE NUMBER OF LIVES SHOWN ON SCREEN
102         this.lives.setText("x" + String.valueOf(lives));
103     }
104 
105     void gameOver(){ //TRIGGER GAME OVER MESSAGE
106         Text gm = new Text(180.0,240.0,"GAME OVER");
107         gm.setFill(Color.WHITE);
108         gm.setFont(Font.font("Tlwg Typo", 50.0));
109         this.pane.getChildren().add(gm);        
110     }
111 }
